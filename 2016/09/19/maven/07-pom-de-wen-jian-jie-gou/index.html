07--pom的文件结构
maven
maven
发布日期:   2016-09-19
阅读次数:  

POM常用标签如下

<!--pom项目根标签，指定一些项目的约束-->
<project>  
    <!--pom的版本，固定值4.0.0-->
    <modelVersion>4.0.0</modelVersion>  

    <!--当前maven工程坐标-->
    <!--组织名-->
    <groupId>com.nenu.test</groupId>
    <!--模块名-->
    <artifactId></artifactId>
    <!--版本-->
    <version></version>
    <!--打包方式，可选值jar,war,pom等-->
    <packaging></packaging>

    <!--描述信息-->
    <!--唯一标示-->
    <id></id>
    <!--工程名字描述-->
    <name></name>
    <!--工程简介-->
    <description></description>

    <!--依赖-->
    <dependencies>
        <dependency>
            <!--这里首先是依赖的库的坐标-->
            <!--作用域：可选值：compile，provided，test，system，runtime-->
            <scope><scope>
            <!--排除依赖-->
            <exclusions>
                <exclusion>
                    <!--坐标不需要版本号-->
                </exclusion>
            </exclusions>
        </dependency>
    </dependencies>

    <!--只用于父类中-->
    <dependencyManagement>
        <dependencies>
            <dependency>
                <!--这里首先是依赖的库的坐标-->
                <!--作用域：可选值：compile，provided，test，system，runtime，import-->
                <scope><scope>
            </dependency>
        </dependencies>
    </dependencyManagement>

    <!--构建时候的一些配置-->
    <build>
        <!--插件-->
        <plugins>
            <plugin>
                <!--也是需要一个坐标信息-->

                <!--执行信息-->
                <executions>
                    <execution>
                        <!--作用阶段-->
                        <phase></phase>
                        <!--目标-->
                        <goals>
                                <goal></goal>
                            </goals>
                    </execution>
                </executions>
            </plugin>
        </plugins>
    </build>

    <!--父模块-->
    <parent>
        <!--坐标信息-->
    </parent>

    <!--用于聚合某些模块-->
    <modules>
        <module>
            <!--坐标信息-->
        </module>
    </modules>
</project>
