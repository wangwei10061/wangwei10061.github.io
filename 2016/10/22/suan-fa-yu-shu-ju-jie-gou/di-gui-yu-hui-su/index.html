  1. 集合的所有子集
    法一： 回溯
    `java
    import java.util.*;

class Solution {
public List<List> subsets(int[] nums) {
List<List> result = new LinkedList<>();
addToSubsets(0, nums, new LinkedList(), result);

    return result;
}

// 采用回溯法，用递归的形式定义
public void addToSubsets(int i, int[] nums, LinkedList<Integer> item, List<List<Integer>> result) {
    if (i >= nums.length) {
        return ;
    }

    // 选择第i个元素，item表示一个单独的子集，result表示子集的集合
    item.add(nums[i]);
    if (!result.contains(item)) {
        result.add((LinkedList<Integer>)(item.clone()));
    }
    addToSubsets(i + 1, nums, item, result);

    // 不选第i个元素
    item.removeLast();
    addToSubsets(i + 1, nums, item, result);

}

}


法二：位运算法  
```java
import java.util.*;

class Solution {
    public List<List<Integer>> subsets(int[] nums) {

        List<List<Integer>> result = new ArrayList<>();

       // 位图法，取nums.length长度的二进制，对应位置为0表示不选择该位，1表示选择该位
        char[] ones = new char[nums.length];
        Arrays.fill(ones, '1');
        int num = Integer.parseInt(new String(ones), 2);

        for (int i = 0; i <= num; i++) {
            String current = String.format("%" + nums.length + "s", Integer.toBinaryString(i)).replace(' ', '0');

            List<Integer> item = new ArrayList<>();
            for (int j = 0; j < current.length(); j++) {
                if (current.charAt(j) == '1') {
                    item.add(nums[j]);
                }
            }
            result.add(item);
        }

        return result;
    }
}

或下面这种方式效率更高

import java.util.*;

class Solution {
    public List<List<Integer>> subsets(int[] nums) {

        List<List<Integer>> result = new ArrayList<>();

       // 位图法，取nums.length长度的二进制，对应位置为0表示不选择该位，1表示选择该位
        int num = 1 << nums.length;

        for (int i = 0; i < num; i++) {
            List<Integer> item = new ArrayList<>();
            for (int j = 0; j < nums.length; j++) {
                if ((i & (1 << j)) != 0) {
                    item.add(nums[j]);
                }
            }
            result.add(item);
        }

        return result;
    }
}

public void addToSubsets(int i, int[] nums, LinkedList<Integer> item, List<List<Integer>> result, int currSum, int target) {
    if (i >= nums.length || currSum > target) {
        return ;
    }

    // 选择第i个元素，item表示一个单独的子集，result表示子集的集合
    currSum += nums[i];
    item.add(nums[i]);

    LinkedList<Integer> temp = (LinkedList<Integer>)(item.clone());
    if (currSum == target && !result.contains(temp)) {
        result.add(temp);
    }
    addToSubsets(i + 1, nums, item, result, currSum, target);

    // 不选第i个元素
    currSum -= nums[i];
    item.removeLast();
    addToSubsets(i + 1, nums, item, result, currSum, target);

3. [合法的括号序列](https://leetcode.com/problems/generate-parentheses/submissions/)
```java
class Solution {
    public List<String> generateParenthesis(int n) {
        List<String> result = new ArrayList<>();
        generateParenthesis(n, n, "", result);

        return result;
    }

    public void generateParenthesis(int left, int right, String item, List<String> result) {
        if (left == 0 && right == 0) {
            result.add(item.toString());
            return ;
        }

        // 只有在还剩余有左括号未插入时才能插入左括号
        if (left > 0) {
            generateParenthesis(left - 1, right, item + "(", result);
        }

        // 只有在插入了左括号之后才能插入右括号
        if (left < right) {
            generateParenthesis(left, right - 1, item + ")", result);
        }
    }
}
