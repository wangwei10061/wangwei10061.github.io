创建图形化用户界面的步骤
  1. 创建Frame对象
  2. 对窗体进行基本的设置：大小，位置等
  3. 定义组件
  4. 将组件add到窗体中
  5. 设置窗体为可见
用事件监听机制实现对应的功能
  • 事件监听基本概念
    • 事件源：awt或swing包中的组件
    • 事件：每一个事件都有自己的对应事件和公共事件
    • 监听器：某一事件对应的监听器封装了该事件可以触发的动作
    • 事件处理：具体的处理需要我们来实现

  • 步骤

  • 注意
    平常开发将图形界面与事件处理分离开来，以提高代码的可读性，
    也使程序后期的修改维护更简单

  • 例程

import java.awt.*;
import java.awt.event.*;

//将图形用户界面和事件的处理分离开来
public class SeparatedAwtListener {

    // 建立引用
    private Frame f;
    private Button b;
    private TextField tf;

    SeparatedAwtListener() {
        init();
    }

    // 初始化：
    public void init() {
        // 界面
        f = new Frame("my frame");

        f.setBounds(200, 300, 500, 400);

        b = new Button("退出");
        f.add(b);
        tf = new TextField(20);
        f.add(tf);

        f.setLayout(new FlowLayout());
        f.setVisible(true);

        myEvent();
    }

    // 事件监听
    public void myEvent() {
        // 窗口事件
        f.addWindowListener(new WindowAdapter() {
            @Override
            public void windowClosing(WindowEvent e) {
                System.exit(0);
            }
        });
        // 动作事件
        b.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                System.out.println("Action");
            }
        });
        // 鼠标事件
        b.addMouseListener(new MouseAdapter() {
            @Override
            public void mouseClicked(MouseEvent e) {
                System.out.println("点击");
            }
        });
        // 键盘事件
        b.addKeyListener(new KeyAdapter() {
            public void keyPressed(KeyEvent e) {
                if (e.isControlDown() && e.getKeyCode() == KeyEvent.VK_ENTER) {
                    System.exit(0);
                }
            }
        });

        tf.addKeyListener(new KeyAdapter() {
            public void keyPressed(KeyEvent e) {
                if (!(e.getKeyCode() >= KeyEvent.VK_0 && e.getKeyCode() <= KeyEvent.VK_9))
                    e.consume();
            }
        });

    }

    public static void main(String[] args) {
        new SeparatedAwtListener();
    }

}

import java.awt.*;
import java.awt.event.*;
import java.io.File;

public class ImitateFileExplorer {

    private Frame f;
    private TextField tf;
    private TextArea ta;
    private Button but;
    private Panel p;
    private Dialog dlg;
    private Button but_dlg;

    ImitateFileExplorer() {
        init();
    }

    private void init() {
        f = new Frame("模拟文件管理器 ");
        tf = new TextField(50);
        but = new Button("显示文件名");
        ta = new TextArea();
        p = new Panel();

        f.setBounds(200, 100, 800, 700);
        f.setLayout(new BorderLayout());

        p.add(tf);
        p.add(but);

        f.add(p, BorderLayout.NORTH);
        f.add(ta, BorderLayout.CENTER);

        f.setVisible(true);

        eventDo();
    }

    public void eventDo() {
        // 关闭窗口
        f.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                System.exit(0);
            }
        });
        // 实现按钮点击对应的功能

        but.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                String path = tf.getText();

                File file = new File(path);

                if (file.exists()) {
                    if (file.isDirectory()) {
                        ta.setText("");
                        String[] fileNameList = file.list();
                        int i = fileNameList.length - 1;
                        System.out.println(i);
                        while (i > 0) {
                            ta.append(fileNameList[i--] + '\n');
                        }

                    } else
                        ta.setText(file.getName());
                } else {
                    // 创建一个对话框提示输入有误
                    dlg = new Dialog(f);
                    dlg.setTitle("出错了");
                    dlg.setBounds(400, 300, 200, 100);

                    dlg.add(new TextField(tf.getText() + "未找到!!!!"));
                    but_dlg = new Button("确定");
                    dlg.add(but_dlg);

                    dlg.setLayout(new FlowLayout());
                    dlg.setVisible(true);

                    doDialogEvent();
                }

            }
        });
    }

    // 对对话框进行事件处理
    public void doDialogEvent() {
        dlg.addWindowListener(new WindowAdapter() {
            public void windowClosing(WindowEvent e) {
                dlg.setVisible(false);
            }
        });
        but_dlg.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                dlg.setVisible(false);
            }
        });
    }

    public static void main(String[] args) {
        new ImitateFileExplorer();
    }

}
